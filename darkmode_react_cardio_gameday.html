User Search

import React, { useState, useEffect } from "react";
import axios from "axios";

interface User {
  id: number;
  name: string;
  email: string;
}

const UserSearch: React.FC = () => {
  const [users, setUsers] = useState<User[]>([]);
  const [filteredUsers, setFilteredUsers] = useState<User[]>([]);
  const [isLoading, setIsLoading] = useState<boolean>(true);

  useEffect(() => {
    try {
      const fetchUsers = async () => {
        const res = await axios.get(
          "https://jsonplaceholder.typicode.com/users",
        );

        setUsers(res.data);
        setFilteredUsers(res.data);
        setIsLoading(false);
      };

      fetchUsers();

    } catch (error) {
      console.log(error);
    }
  }, []);

  const handleFilterChange = (input: string) => {
    const query = input.toLowerCase().trim();

    if (!query) {
        setFilteredUsers(users);
    }

    setFilteredUsers(users.filter(({name, email}) => {
        return name.toLowerCase().trim().includes(query) || email.toLowerCase().trim().includes(query);
    }));
  }

  return (
    <>
      <h1>User Search</h1>

        <label htmlFor="filter">Filter by Name</label>
        <input type="text" id="filter" name="filter" onChange={(e) => handleFilterChange(e.target.value)}/>

      {isLoading && <p>Loading users...</p>}

      <h2>Users:</h2>

      {filteredUsers.length === 0 && <p>No users available</p>}

      <ul>
        {filteredUsers.map(({ id, name, email }) => {
          return (
            <li key={id}>
              {name} - {email}
            </li>
          );
        })}
      </ul>
    </>
  );
};

export default UserSearch;

String List Form

import React, { useState } from "react";

const StringListForm: React.FC = () => {
  const [stringList, setStringList] = useState<string[]>([]);

  const [value, setValue] = useState<string>("");

  const handleFormSubmit = (e: any) => {
    e.preventDefault();

    setStringList((prev) => [...prev, value]);

    setValue("");
  };

  return (
    <>
      <h1>String List with Form</h1>

      <h2>String Count: {stringList.length}</h2>

      <form>
        <label htmlFor="input">Input: </label>
        <input
          type="text"
          id="input"
          onChange={(e) => setValue(e.target.value)}
        />
        <button type="submit" onClick={(e) => handleFormSubmit(e)}>
          Submit
        </button>
      </form>

      <h2>String List:</h2>
      <ul>
        {stringList.map((str) => (
          <li>{str}</li>
        ))}
      </ul>
    </>
  );
};

export default StringListForm;
