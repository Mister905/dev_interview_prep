Return Common Values Between Arrays

Return values that appear in both arrays without nested loops.

Show Solution 
const return_common_values = (arr1, arr2) => {
  const value_map = new Set(arr1);
  const result = [];
  for (let val of arr2) {
    if (value_map.has(val)) {
      result.push(val);
      value_map.delete(val); 
    }
  }
  return result;
};

// Example:
console.log(return_common_values([1,2,3,4], [3,4,5,6])); // [3,4]

Output: true

const isPalindrome = (str) => {
  let reversed_str = "";
  for (let i = str.length - 1; i >= 0; i--) {
    reversed_str += str[i];
  }

Max Character

Find the character that appears most frequently and return it along with its count. 

Input: "abccccccd"
Output: "c: 6"
Show Solution 
const max_character = (str) => {
  let char_map = {};
  for (let char of str.split("")) {
    if (char_map[char]) char_map[char]++;
    else char_map[char] = 1;
  }
  let max_char = "";
  let max_char_count = 0;
  for (let char in char_map) {
    if (char_map[char] > max_char_count) {
      max_char = char;
      max_char_count = char_map[char];
    }
  }
  return `${max_char}: ${max_char_count}`;
};

Anagrams

Check if two strings are anagrams of each other (ignoring spaces and case). 

Input: "rail safety", "fairy tales"
Output: true

const create_char_map = (str) => {
  let str_array = str.split("");
  let char_map = {};
  for (let char of str_array) {
    if (char_map[char]) char_map[char]++;
    else char_map[char] = 1;
  }
  return char_map;
};

const anagrams = (str1, str2) => {
  let sanitized_str1 = str1.replace(/\W/g, "").toLowerCase();
  let sanitized_str2 = str2.replace(/\W/g, "").toLowerCase();
  let char_map_1 = create_char_map(sanitized_str1);
  let char_map_2 = create_char_map(sanitized_str2);
  if (Object.keys(char_map_1).length !== Object.keys(char_map_2).length) return false;
  for (let char in char_map_1) {
    if (char_map_1[char] !== char_map_2[char]) return false;
  }
  return true;
};

Longest Substring Without Repeating Characters

Given a string, find the length of the longest substring without repeating characters. 

Input: "abcabcbb"
Output: 3
Show Solution 
const longest_substring_without_repeating = (str) => {
  const char_map = {};
  let max_length = 0;
  let start = 0;
  for (let i = 0; i < str.length; i++) {
    const current_char = str[i];
    if (current_char in char_map && char_map[current_char] >= start) {
      start = char_map[current_char] + 1;
    }
    char_map[current_char] = i;
    max_length = Math.max(max_length, i - start + 1);
  }
  return max_length;
};
Algorithmic Variations

Two Sum

Given an array of integers and a target number, return the indices of the two numbers that add up to the target. 

Input: [1, 4, 6, 3], target = 7
Output: [0, 3]
Show Solution 
const two_sum = (arr, target) => {
  const int_map = new Map();
  for (let i = 0; i < arr.length; i++) {
    const current = arr[i];
    const required_value = target - current;
    if (int_map.has(required_value)) {
      return [int_map.get(required_value), i];
    }
    int_map.set(current, i);
  }
  return null;
};

Two Pointer - Pair Sum

Check if any two numbers sum to target in sorted array.

Show Solution 
const pair_sum = (arr, target) => {
  let left=0, right=arr.length-1;
  while(left<right){
    const sum = arr[left]+arr[right];
    if(sum===target) return [arr[left], arr[right]];
    else if(sum < target) left++;
    else right--;
  }
  return null;
};

Two Pointer - Remove Duplicates Sorted

Remove duplicates from sorted array in-place.

const remove_duplicates_sorted = (arr) => {
  if(arr.length===0) return 0;
  let i=0;
  for(let j=1;j<arr.length;j++){
    if(arr[j]!==arr[i]){
      i++;
      arr[i]=arr[j];
    }
  }
  return i+1; // length of unique array
};

Sliding Window - Max Sum of Subarray

Find maximum sum of contiguous subarray of size k.

Show Solution 
const max_subarray_sum = (arr, k) => {
  let maxSum = 0, windowSum = 0;
  for (let i=0; i<k; i++) windowSum += arr[i];
  maxSum = windowSum;
  for (let i=k; i<arr.length; i++) {
    windowSum += arr[i] - arr[i-k];
    maxSum = Math.max(maxSum, windowSum);
  }
  return maxSum;
};

Sliding Window - Longest Substring K Distinct

Longest substring with at most K distinct characters.

Show Solution 
const longest_substring_k_distinct = (s, k) => {
  let map = {}, start=0, maxLen=0;
  for (let end=0; end<s.length; end++){
    map[s[end]] = (map[s[end]]||0)+1;
    while(Object.keys(map).length > k){
      map[s[start]]--;
      if(map[s[start]]===0) delete map[s[start]];
      start++;
    }
    maxLen = Math.max(maxLen, end-start+1);
  }
  return maxLen;
};

Filter Array on Server

Filter array based on query parameters.

// Expected input: GET request to /api/filter?min=3
// Expected output: JSON array of numbers >= min
// Example: /api/filter?min=3 -> [3,4,5]
Show Solution 
app.get('/api/filter', (req, res) => {
  const items = [1,2,3,4,5];
  const min = parseInt(req.query.min) || 0;
  const filtered = items.filter(item => item >= min);
